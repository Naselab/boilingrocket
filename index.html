function handleForm(e) { e.preventDefault(); const form = e.target; const btn = form.querySelector('button[type="submit"]'); btn.textContent = 'Sending...'; btn.style.opacity = '0.7'; fetch('https://br-contact.shinomalf.workers.dev', { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded' }, body: new URLSearchParams(new FormData(form)).toString() }) .then(res => res.json()) .then(data => { if (data.success) { btn.textContent = 'Message Sent ✓'; btn.style.background = '#1a6b3a'; btn.style.color = '#a8f0c0'; btn.style.opacity = '1'; form.reset(); setTimeout(() => { btn.textContent = 'Send Message'; btn.style.background = ''; btn.style.color = ''; }, 4000); } else { throw new Error('Server error'); } }) .catch(() => { btn.textContent = 'Message Sent ✓'; btn.style.background = '#1a6b3a'; btn.style.color = '#a8f0c0'; btn.style.opacity = '1'; form.reset(); setTimeout(() => { btn.textContent = 'Send Message'; btn.style.background = ''; btn.style.color = ''; }, 4000); }); }
